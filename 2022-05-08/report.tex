根据该表写出逻辑表达式\eqref{eqn:exct2}

\begin{equation}
	\label{eqn:exct2}
	\begin{aligned}
		D_3&=\ols{Q_3}\cdot Q_2\cdot Q_1 \\
		D_2&=\ols{Q_3}\cdot\ols{Q_2}\cdot Q_1 + \ols{Q_3}\cdot Q_2\cdot\ols{Q_1} \\
		&= \ols{\ols{\ols{Q_3}\cdot(Q_2 \oplus Q_1)}} \\
		D_1&=\ols{Q_3}\cdot\ols{Q_2}\cdot\ols{Q_1} \\
	\end{aligned}
\end{equation}
